21 int write_wave(const char filename[],
22  int16_t data[], int size, int sample_rate)
23 {
24  ofstream outfile (filename,ofstream::binary);
25  uint32_t chunksize;
26  uint32_t subchunk1size;
27  uint32_t subchunk2size;
28  uint16_t audioformat;
29  uint16_t numchannels;
30  uint32_t samplerate;
31  uint32_t byterate;
32  uint16_t blockalign;
33  uint16_t bitspersample;
34 
35  if(!outfile.good())
36  {
37  cout << "convert_to_wave: Unable to open file \"";
38  cout << filename;
39  cout << "\" for output." << endl;
40  cout<< strerror( errno ) << "\n" << endl;
41  return 0;
42  }
43  subchunk1size = 16; // always 16 for PCM data
44  audioformat = 1; // this is the code for PCM
45  numchannels = 1;
46  samplerate = sample_rate;
47  bitspersample = sizeof(int16_t)*8;
48  byterate = samplerate * numchannels * bitspersample/8;
49  blockalign = numchannels * bitspersample/8;
50  subchunk2size = size * numchannels * sizeof(int16_t);
51  chunksize = 4 + (8 + subchunk1size) + (8 + subchunk2size);
52 
53  outfile.write("RIFF",4);
54  outfile.write((char*)&chunksize,4);
55  outfile.write("WAVE",4);
56 
57  outfile.write("fmt ",4);
58  outfile.write((char*)&subchunk1size,sizeof(subchunk1size));
59  outfile.write((char*)&audioformat,sizeof(audioformat));
60  outfile.write((char*)&numchannels,sizeof(numchannels));
61  outfile.write((char*)&samplerate,sizeof(samplerate));
62  outfile.write((char*)&byterate,sizeof(byterate));
63  outfile.write((char*)&blockalign,sizeof(blockalign));
64  outfile.write((char*)&bitspersample,sizeof(bitspersample));
65 
66  outfile.write("data",4);
67  outfile.write((char*)&subchunk2size,sizeof(subchunk2size));
68  outfile.write((char*)data,size * sizeof(int16_t));
69  if(!outfile.good())
70  {
71  cout << "convert_to_wave: Error writing to \"";
72  cout << filename;
73  cout << "\"." << endl;
74  cout<< strerror( errno ) << endl;
75  return 0;
76  }
77  outfile.close();
78  return 1;
79 }

92  char yn;

122  // count how many numbers we can read from the file
123  size = 0;

129  input >> tmp;
130  if(fabs(tmp)>max)
131  max = fabs(tmp);
132  }

134  input.clear();
135  // go back to the beginning of the file
136  input.seekg(ios_base::beg);

143  return 0;
144  }

146  {
147  cout << "convert_to_wave: Input file \"";
148  cout << input_filename;
149  cout << "\" is empty or does not contain only numbers." << endl;
150  return 0;
151  }
152  // allocate enough space to store all of the numbers
153  data16 = new(nothrow) int16_t[size];

155  {
156  cout<<"convert_to_wave: Memory allocation failed."<<endl;
157  cout<< strerror( errno ) << endl;
158  return 0;
159  }
160  // read them all in as floats, then convert to 16 bit ints
161  for(i=0;i<size;i++)
162  {
163  input >> tmp;
164  data16[i] = (int16_t)((tmp/max) * (SHRT_MAX-1));
165  }

167  {
168  cout << "convert_to_wave: Error reading input file \"";

171  cout<< strerror( errno ) << endl;
172  return 0;
173  }
174  input.close();

176  delete[] data16;
